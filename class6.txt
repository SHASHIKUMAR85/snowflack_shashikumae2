select A.*,'|',B.*  from employees A  inner join 
                         dependents B
                        on A.EMPLOYEE_ID = B.EMPLOYEE_ID

    ,A.first_name

    ,A.salary
    ,B.FIRST_NAME
    ,B.RELATIONSHIP
    from employees A  ,
        dependents B
        where A.EMPLOYEE_ID = B.EMPLOYEE_ID
